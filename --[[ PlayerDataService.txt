local DataStoreService = game:GetService("DataStoreService")
local Players = game:GetService("Players")
local DataStore = DataStoreService:GetDataStore("PlayerDepartmentData_V2")

local PlayerDataService = {}
local PlayerData = {}

local DEFAULT_DATA = {
	Department = "Rig Security",
	XP = {
		["Rig Security"] = 0,
		["Rig Maintenance Team"] = 0,
		["Logistical Operations Team"] = 0,
		["Rig Operations Team"] = 0,
		["Ground Service Personnel"] = 0,
		Corporate = 0,
		["Rig Administration"] = 0,
	},
	RankIndex = 1,
}

local function Copy(tbl)
	local copy = {}
	for k, v in pairs(tbl) do
		copy[k] = type(v) == "table" and Copy(v) or v
	end
	return copy
end

function PlayerDataService:LoadPlayer(player)
	local success, saved = pcall(function()
		return DataStore:GetAsync(player.UserId)
	end)

	local data
	if success and saved then
		data = Copy(DEFAULT_DATA)
		data.Department = saved.Department or data.Department
		data.RankIndex = saved.RankIndex or data.RankIndex
		if saved.XP then
			for dept, xp in pairs(saved.XP) do
				data.XP[dept] = xp
			end
		end
	else
		data = Copy(DEFAULT_DATA)
	end

	PlayerData[player] = data
	return data
end

function PlayerDataService:SavePlayer(player)
	local data = PlayerData[player]
	if not data then
		return
	end

	local toSave = {
		Department = data.Department,
		RankIndex = data.RankIndex,
		XP = {},
	}
	for dept, xp in pairs(data.XP) do
		toSave.XP[dept] = xp
	end

	pcall(function()
		DataStore:SetAsync(player.UserId, toSave)
	end)
end

function PlayerDataService:Get(player)
	return PlayerData[player]
end

function PlayerDataService:Remove(player)
	PlayerData[player] = nil
end

Players.PlayerRemoving:Connect(function(player)
	PlayerDataService:SavePlayer(player)
	PlayerDataService:Remove(player)
end)

game:BindToClose(function()
	for player in pairs(PlayerData) do
		PlayerDataService:SavePlayer(player)
	end
end)

return PlayerDataService
