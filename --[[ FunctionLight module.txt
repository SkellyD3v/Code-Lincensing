local LightFunctions = {}

local function getLightsFromFolder(folderName)
	local folder = workspace:FindFirstChild(folderName, true)
	if not folder then
		warn("Light folder '" .. folderName .. "' not found.")
		return {}
	end

	local foundLights = {}

	for _, obj in ipairs(folder:GetDescendants()) do
		if obj:IsA("Light") then
			table.insert(foundLights, obj)
		end
	end

	return foundLights
end

function LightFunctions.On(instance)
	local folderName = instance:GetAttribute("LightGroup")
	if not folderName then return end

	for _, light in ipairs(getLightsFromFolder(folderName)) do
		light.Enabled = true
	end
end

function LightFunctions.Off(instance)
	local folderName = instance:GetAttribute("LightGroup")
	if not folderName then return end

	for _, light in ipairs(getLightsFromFolder(folderName)) do
		light.Enabled = false
	end
end

return LightFunctions
