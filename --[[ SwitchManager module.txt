local SwitchBase = require(script.Parent.SwitchBase)
local CollectionService = game:GetService("CollectionService")

local SwitchManager = {}
SwitchManager.SwitchObjects = {}


function SwitchManager:InitializeSwitch(instance)
	local config = {}
	
	-- config stuff
	config.Locked = instance:GetAttribute("Locked") or false
	config.ButtonDelay = instance:GetAttribute("ButtonDelay") or 4
	local target
	if instance.Parent:FindFirstChild("Target") then
		target = instance.Parent.Target.CFrame
	else
		target = instance.CFrame * CFrame.new(0, 0, -1)
	end
	config.TargetCFrame = target
	
	-- running function
	local functionName = instance:GetAttribute("FunctionModule")
	if functionName then
		config.FunctionModule = require(script.Parent[functionName])
	end
	
	local switch = SwitchBase.new(instance, config)
	self.SwitchObjects[instance] = switch
end

function SwitchManager:GetSwitch(instance)
	return self.SwitchObjects[instance]
end

return SwitchManager
