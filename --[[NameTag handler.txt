local Players = game:GetService("Players")
local replicatedStorage = game:GetService("ReplicatedStorage")
local ServerStorage = game:GetService("ServerStorage")

local shared_folder = replicatedStorage:FindFirstChild("Shared")
local dataman = shared_folder:FindFirstChild("DataManagement")


local PlayerDataService = require(dataman.PlayerDataService)
local DepartmentHandler = require(dataman.DepartmentHandler)

local NameTagTemplate = ServerStorage:WaitForChild("NameTag")
local UpdateEvent = ServerStorage.NameTagUpdate

--// functions

local function SetupNameTag(player, character)
	local head = character:WaitForChild("Head")

	local old = head:FindFirstChild("NameTag")
	if old then old:Destroy() end

	local tag = NameTagTemplate:Clone()
	tag.Name = "NameTag"
	tag.Parent = head
	tag.Adornee = head

	local data = PlayerDataService:Get(player)

	tag.PlayerName.Text = player.Name
	tag.Team.Text = data.Department
	tag.TeamRank.Text = DepartmentHandler:GetRankName(data)

	if player.Team then
		tag.Team.TextColor3 = player.Team.TeamColor.Color
	end
end

local function UpdateNameTag(player)
	if not player then return end
	if not player.Character then return end
	local head = player.Character:FindFirstChild("Head")
	if not head then return end
	local tag = head:FindFirstChild("NameTag")
	if not tag then return end
	local data = PlayerDataService:Get(player)
	if not data then return end
	tag.Team.Text = data.Department
	tag.TeamRank.Text = DepartmentHandler:GetRankName(data)
	if player.Team then
		tag.Team.TextColor3 = player.Team.TeamColor.Color
	end
end

UpdateEvent.Event:Connect(UpdateNameTag)

Players.PlayerAdded:Connect(function(player)

	local data = PlayerDataService:LoadPlayer(player)

	player.CharacterAdded:Connect(function(character)
		task.wait(0.5)

		DepartmentHandler:SetDepartment(player, data, data.Department)
		DepartmentHandler:AssignTeam(player, data.Department)

		SetupNameTag(player, character)
	end)
end)
